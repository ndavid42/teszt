export class VizsgafeladatComponent {

  htmlModul!: number;
  bootstrapModul!: number;
  javaScriptModul!: number;
  typeScriptModul!: number;
  angularModul!: number;
  serverModul!: number;

  osszpontszam!: number;

  EredmenyMentes() {
    if (this.htmlModul == null || this.bootstrapModul == null || this.javaScriptModul == null || this.typeScriptModul == null || this.angularModul == null || this.serverModul == null) {
      alert("Az összes mező kitöltése kötelező!");
    }
    else {
      this.osszpontszam = this.htmlModul + this.bootstrapModul + this.javaScriptModul + this.typeScriptModul + this.angularModul + this.serverModul;
      if (this.osszpontszam < 50) {
        this.megoldasok.push("Sikertelen vizsga, szerzett pont: " + this.osszpontszam);
      }
      else if (this.osszpontszam < 60) {
        this.megoldasok.push("Sikeres vizsga (2-es), szerzett pont: " + this.osszpontszam);
      }
      else if (this.osszpontszam < 70) {
        this.megoldasok.push("Sikeres vizsga (3-as), szerzett pont: " + this.osszpontszam);
      }
      else if (this.osszpontszam < 80) {
        this.megoldasok.push("Sikeres vizsga (4-es), szerzett pont: " + this.osszpontszam);
      }
      else {
        this.megoldasok.push("Sikeres vizsga (5-ös), szerzett pont: " + this.osszpontszam);
      }
    }
  }

  megoldasok: string[] = [];
}
